#### 5/28~29 — 핵심 기능 복구

**A-4. BUG-7 건설 시스템 복구** `BuildSystem.cpp` L42~60
> 현재 `"barricade_item"` key로 재료 체크 → items.json에 없어서 건설이 항상 실패

```cpp
// 수정안: crafting_recipes 기반으로 재료 소모
// 간단하게 인벤토리에서 실제 재료를 차감하도록 변경

// Barricade: scrap_metal 2 + plank 2
if (type == BuildingType::Barricade) {
    if (!consumeItem(inv, "scrap_metal", 2)) return Entity{NULL_ENTITY};
    if (!consumeItem(inv, "plank", 2))       return Entity{NULL_ENTITY};
}
// Turret: electronic_part 2 + oil 2 + scrap_metal 3
else if (type == BuildingType::Turret) {
    if (!consumeItem(inv, "electronic_part", 2)) return Entity{NULL_ENTITY};
    if (!consumeItem(inv, "oil", 2))             return Entity{NULL_ENTITY};
    if (!consumeItem(inv, "scrap_metal", 3))     return Entity{NULL_ENTITY};
}
// Workbench: plank 2
else if (type == BuildingType::Workbench) {
    if (!consumeItem(inv, "plank", 2)) return Entity{NULL_ENTITY};
}

// 헬퍼 람다 (함수 상단에 추가)
auto consumeItem = [](InventoryComponent* inv, const char* key, int qty) -> bool {
    if (!inv) return true; // 인벤 없으면 무조건 통과 (테스트용)
    for (int i = 0; i < INVENTORY_GRID_SLOTS; ++i) {
        if (inv->slots[i].isValid() && inv->slots[i].key == key) {
            if (inv->slots[i].quantity < qty) return false;
            inv->slots[i].quantity -= qty;
            if (inv->slots[i].quantity == 0) inv->removeItem(i);
            return true;
        }
    }
    return false;
};
```
